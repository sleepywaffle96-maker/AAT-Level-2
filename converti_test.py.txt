HTML_CALC_TEMPLATE = """
        function verifyCurrentInline() {
            const activeCard = document.getElementById("card_task_" + currentActiveIndex);
            const correctVal = activeCard.getAttribute("data-correct");
            const rA = document.getElementById("opt_" + currentActiveIndex + "_A");
            const rB = document.getElementById("opt_" + currentActiveIndex + "_B");
            const rC = document.getElementById("opt_" + currentActiveIndex + "_C");
            const mA = document.getElementById("mark_" + currentActiveIndex + "_A");
            const mB = document.getElementById("mark_" + currentActiveIndex + "_B");
            const mC = document.getElementById("mark_" + currentActiveIndex + "_C");

            if(mA) mA.style.display = "none";
            if(mB) mB.style.display = "none";
            if(mC) mC.style.display = "none";

            if (rA && rA.checked && mA) {
                mA.style.display = "block";
                mA.innerHTML = correctVal === "A" ? '<span class="feedback-correct">✔️</span>' : '<span class="feedback-wrong">❌</span>';
            } else if (rB && rB.checked && mB) {
                mB.style.display = "block";
                mB.innerHTML = correctVal === "B" ? '<span class="feedback-correct">✔️</span>' : '<span class="feedback-wrong">❌</span>';
            } else if (rC && rC.checked && mC) {
                mC.style.display = "block";
                mC.innerHTML = correctVal === "C" ? '<span class="feedback-correct">✔️</span>' : '<span class="feedback-wrong">❌</span>';
            }
        }

        function resetCurrentForm() {
            const activeCard = document.getElementById("card_task_" + currentActiveIndex);
            if(document.getElementById("mark_" + currentActiveIndex + "_A")) document.getElementById("mark_" + currentActiveIndex + "_A").style.display = "none";
            if(document.getElementById("mark_" + currentActiveIndex + "_B")) document.getElementById("mark_" + currentActiveIndex + "_B").style.display = "none";
            if(document.getElementById("mark_" + currentActiveIndex + "_C")) document.getElementById("mark_" + currentActiveIndex + "_C").style.display = "none";
            const rows = activeCard.querySelectorAll(".option-row");
            rows.forEach(r => r.classList.remove("reveal-correct"));
            ['A', 'B', 'C'].forEach(l => {
                let rb = document.getElementById("opt_" + currentActiveIndex + "_" + l);
                if(rb) rb.checked = false;
            });
        }

        let isCalcBodyExpanded = false;
        let calcCurrentExpressionString = "";
        function toggleCalculatorView() {
            const body = document.getElementById("calcWidgetBody");
            const arrow = document.getElementById("calcToggleArrow");
            isCalcBodyExpanded = !isCalcBodyExpanded;
            body.style.display = isCalcBodyExpanded ? "block" : "none";
            arrow.textContent = isCalcBodyExpanded ? "▼" : "▲";
        }
        function pushCalcChar(char) {
            const display = document.getElementById("calcDisplayScreen");
            if (display.value === "0" && char !== ".") { calcCurrentExpressionString = char; }
            else { calcCurrentExpressionString += char; }
            display.value = calcCurrentExpressionString;
        }
        function clearCalcScreen() { calcCurrentExpressionString = ""; document.getElementById("calcDisplayScreen").value = "0"; }
        function pushCalcFunction(type) {
            const display = document.getElementById("calcDisplayScreen");
            let val = parseFloat(display.value) || 0;
            if (type === 'sqrt') { if (val < 0) { display.value = "Error"; calcCurrentExpressionString = ""; return; } val = Math.sqrt(val); }
            else if (type === 'pow') { val = Math.pow(val, 2); }
            else if (type === 'percent') { val = val / 100; }
            calcCurrentExpressionString = Number(val.toFixed(6)).toString();
            display.value = val;
        }
        function executeCalcExpression() {
            const display = document.getElementById("calcDisplayScreen");
            if (!calcCurrentExpressionString) return;
            try {
                let res = new Function('return (' + calcCurrentExpressionString + ')')();
                calcCurrentExpressionString = Number(res.toFixed(6)).toString();
                display.value = calcCurrentExpressionString;
            } catch (e) { display.value = "Error"; calcCurrentExpressionString = ""; }
        }
    </script>
</body>
</html>
"""

folder = "level-2"
if os.path.exists(folder):
    for file_name in os.listdir(folder):
        if file_name.endswith(".html"):
            path = os.path.join(folder, file_name)
            with open(path, "r", encoding="utf-8") as f:
                html = f.read()

            cards = re.findall(r'(<div class="test-card-container".*?</div>\s*</div>)', html, re.DOTALL)
            
            if cards:
                cleaned_cards = []
                for idx, card in enumerate(cards):
                    card = re.sub(r'id="card_task_\d+"', f'id="card_task_{idx}"', card)
                    card = re.sub(r'name="ans_\d+"', f'name="ans_{idx}"', card)
                    card = re.sub(r'id="opt_\d+_(.*?)"', r'id="opt_' + str(idx) + r'_\1"', card)
                    card = re.sub(r'id="mark_\d+_(.*?)"', r'id="mark_' + str(idx) + r'_\1"', card)
                    card = re.sub(r'onclick="document\.getElementById\(\'opt_\d+_(.*?)\'\)\.checked\s*=\s*true"', r"onclick=\"document.getElementById('opt_" + str(idx) + r"_\1').checked = true\"", card)
                    
                    if idx == 0:
                        card = card.replace('test-card-container', 'test-card-container active')
                    else:
                        card = card.replace('test-card-container active', 'test-card-container')
                    cleaned_cards.append(card)

                    new_content = HTML_HEAD_TEMPLATE + "\n".join(cleaned_cards) + HTML_FOOTER_TEMPLATE + HTML_JS_TEMPLATE + HTML_CALC_TEMPLATE

                
                with open(path, "w", encoding="utf-8") as f:
                    f.write(new_content)
                print(f"✅ Migrato con successo lo stile Osborne per: {file_name}")
